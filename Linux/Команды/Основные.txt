
pwd - выдаёт директорию где ты находишься

ls - показать файлы в каталаге

ls -l - показать файлы с датой создания, его размером, автором и правами

ls -a - показать директорию вместе со скрытыми файлами

cd <Путь> - перейти в нужную директорию

cd .. - вернутся назад по пути

cd ../.. - поднятся на два уровня вверх

mkdir <название директории> - создание директории

touch <название файла> - создание файла

nano <название файла> - редактирование файла

cp <что скопировать> <куда скопировать> - копирование файла можно переименовать файл если при куда скопировать последними символами
указать имя
cp 1.png pictures/2.png

cp -r <путь к папке> <путь к новому месту> - копировать папку рекурсивно

mv <название файла> <куда перенести> - перенести файл(не копирование)

rm * - удалить все файлы из директории

rm <имя файла> - удалить нужный файл

rm -r <папка> - удаляет всё вместе с файлами

rm -rf <папка> - удалить все файлы без лишних вопросов(вы уверены что хотите удалить и тд)
