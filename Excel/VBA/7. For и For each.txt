Цикл for - пошаговое значение переменных в диапазоне.

Синтаксис цикла for

Sub testSub()
    Dim i As Long
    'Тут нужно объявить количество шагов - в данном цикле от одного до 10 включительно    
    For i = 1 To 10
        'Поочерёдно проходим каждую ячейку и проставляем значение i
        ThisWorkbook.Worksheets(1).Cells(i, 1) = i
        MsgBox "Значение = " & i
    'Этой строчкой заканчивается тело цикла for, то есть мы просто переключается на следующее i
    Next i
      
End Sub

Цикл for each

Sub testSub2()
    
    'Объявляем переменную, в которую будем записывать перебираемые значения, в данном случае нужно обратить внимание на тип
    Dim cellChecked As Range
    
    'Объявляем цикл и диапазон в котором будем что-то делать
    For Each cellChecked In Range("A1:A10")
        MsgBox "Function testSub2 " & cellChecked.Value
    'Переходим к следующей ячейки
    Next cellChecked
    
End Sub

Если мы выберем диапазон в котором будет более одного столбика, то мы будем в начале перебирать строку, а затем переходить к 
следующему столбцу то есть, слева направо и сверху вниз.
Чтобы подробнее посмотреть в каком порядке мы проходим можно скомпилировать следующий код

Sub testSub2()

    Dim cellChecked As Range
    Dim i As Long
    
    i = 1
    
    For Each cellChecked In Range("A1:C10")
        
        cellChecked.Value = i
        
        i = i + 1
        
    Next cellChecked
    
End Sub
