Для смены языка в админке 

меняем переменную

LANGUAGE_CODE = 'ru-ru'

для того чтоб сменить заголовок в админке нужно в проекте в файле urls.py добавить строчку

admin.site.site_header = 'Как хотим назвать админку'

Для более мелкого заголовка там же пишем

admin.site.index_title = 'Моя супер админка'

Для настройки отображения нужных столбцов таблиц и редактирования их в admin.py
также можно выводить сразу отсортированные поля

@admin.register(Movie)
class MovieAdmin(admin.ModelAdmin):
    list_display = ['name', 'rating', 'year', 'budget'] # вывод полей
    list_editable = ['rating', 'year', 'budget'] # редактирование полей
# мы не можем менять первое поле из тех что указали, так как первое поле всегда является ссылкой
    ordering = ['rating'] #по какому полю сортировать
    list_per_page = 3 #пагинация
